CREATE PROCEDURE [dbo].[sp_GetUsers]
AS 
BEGIN

INNER JOIN UserRoles UR ON UR.UserId=U.UserId

CREATE PROCEDURE [dbo].[sp_LoginUser] 
(
@Email nvarchar(100),
@PasswordHash NVARCHAR(200)
)
AS
BEGIN
SET NOCOUNT ON
SELECT U.UserId,U.FullName,U.Phone,U.Status,R.RoleName FROM Users U
INNER JOIN UserRoles UR ON UR.UserId=U.UserId
INNER JOIN Roles R ON R.RoleId=UR.RoleId
WHERE U.Email=@Email AND U.PasswordHash=@PasswordHash AND U.Status=1
END
GO

CREATE PROCEDURE [dbo].[sp_RegisterUser]
(
@Email nvarchar(100),
@PasswordHash nvarchar(200),
@FullName nvarchar(150),
@Phone nvarchar(10),
@RoleId int,
@IpAddress nvarchar(50)
)
AS
BEGIN
SET NOCOUNT ON
IF EXISTS (SELECT 1 FROM Users WHERE Email=@Email)
BEGIN
RAISERROR('Email Already Exists',16,1)
RETURN
END
INSERT INTO Users(Email,PasswordHash,FullName,Phone,IpAddress)
VALUES (@Email,@PasswordHash,@FullName,@Phone,@IpAddress)
DECLARE @UserId int = Scope_Identity()
INSERT INTO UserRoles(UserId,RoleId,IpAddress)
VALUES (@UserId,@RoleId,@IpAddress)
END
